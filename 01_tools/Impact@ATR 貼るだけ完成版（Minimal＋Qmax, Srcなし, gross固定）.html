Google Apps Script（任意：Webhook 受け側）

以下を Google Apps Script に貼り付け、デプロイ→ウェブアプリにして URL を上の Webhook に設定してください。

// Google Apps Script（簡易ログ収集：ヘッダー追従・追記）
// 1) SPREADSHEET_ID を自分のシートIDに置換
// 2) SECRET を任意文字列で設定（クライアント側と一致させる）
// 3) デプロイ → ウェブアプリ（全員アクセス可）

const SPREADSHEET_ID = 'YOUR_SPREADSHEET_ID';
const SHEET_NAME = 'ATR Logs';
const SECRET = 'CHANGE_ME_SECRET';

function doOptions(e){ return _cors(200,{ok:true}); }
function doGet(e){ return _cors(200,{ok:true}); }

function doPost(e){
  try{
    const body = JSON.parse(e.postData.contents || '{}');
    if(body.secret !== SECRET) return _cors(403,{ok:false,error:'forbidden'});

    const header = Array.isArray(body.header) ? body.header : null;
    const rows = Array.isArray(body.rows) ? body.rows : [];
    if(!header || !header.length || !rows.length) return _cors(400,{ok:false,error:'invalid payload'});

    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sh = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME);

    const lastRow = sh.getLastRow();
    if(lastRow === 0){
      sh.getRange(1,1,1,header.length).setValues([header]);
      sh.getRange(2,1,rows.length,header.length).setValues(rows);
      return _cors(200,{ok:true, inserted:rows.length, headerWritten:true});
    }else{
      const existingHeader = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0];
      const same = existingHeader.length===header.length && existingHeader.every((v,i)=>String(v)===String(header[i]));
      let startRow = sh.getLastRow()+1;
      if(!same){
        sh.getRange(startRow,1,1,1).setValue('');
        startRow += 1;
        sh.getRange(startRow,1,1,header.length).setValues([header]);
        startRow += 1;
      }
      sh.getRange(startRow,1,rows.length,header.length).setValues(rows);
      return _cors(200,{ok:true, inserted:rows.length, headerChanged:!same});
    }
  }catch(err){
    return _cors(500,{ok:false,error:String(err)});
  }
}

function _cors(status,obj){
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON)
    .setHeader('Access-Control-Allow-Origin','*')
    .setHeader('Access-Control-Allow-Methods','POST, GET, OPTIONS')
    .setHeader('Access-Control-Allow-Headers','Content-Type');
}
